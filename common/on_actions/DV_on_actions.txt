on_actions = {

	#Communist Countries Red
	on_government_change = {
			effect = {
				#Make Countries Red if Communism
				#China
				if = {
					limit = {
						original_tag = CHI
						has_government = communism
					}
					set_cosmetic_tag = PRC_Cosmetic
				}
				#Korea
				if = {
					limit = {
						original_tag = KOR
						has_government = communism
					}
					set_cosmetic_tag = DPK_Cosmetic
				}
				#Russia
				if = {
					limit = {
						original_tag = SOV
						has_government = communism
						has_cosmetic_tag = RUS_Cosmetic
					}
					drop_cosmetic_tag = yes
				}
				#Remove Red color not communist
				#China
				if = {
					limit = {
						original_tag = CHI
						NOT = { has_government = communism }
						has_cosmetic_tag = PRC_Cosmetic
					}
					drop_cosmetic_tag = yes
				}
				#Korea
				if = {
					limit = {
						original_tag = KOR
						NOT = { has_government = communism }
						has_cosmetic_tag = DPK_Cosmetic
					}
					drop_cosmetic_tag = yes
				}
				#Russia
				if = {
					limit = {
						original_tag = SOV
						NOT = { has_government = communism }
					}
					set_cosmetic_tag = RUS_Cosmetic
				}
			}
	}
	
	#ROOT is subject FROM is previous overlord
	on_subject_free = {
		effect = {
			if = {
				limit = { 
					original_tag = CRI
				} 
				drop_cosmetic_tag = yes
			}
		}
	}
	
	#used when puppeting in a peace conference
	#ROOT = nation being puppeted, FROM = overlord
	on_puppet = {
		effect = {
			if = {
				limit = { 
					AND = { 
						original_tag = CRI
						is_subject_of = GER
						}
					 }
				set_cosmetic_tag = CRI_GER
			}
		}
	}

}
